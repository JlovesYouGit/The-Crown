# AMD Quantum Rendering Service CMakeLists.txt
# Builds the Windows service for background quantum rendering

cmake_minimum_required(VERSION 3.10)

# Project information
project(QuantumRenderService VERSION 1.0.0 DESCRIPTION "AMD Quantum Rendering Background Service")

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add the service executable
add_executable(QuantumRenderService
    QuantumRenderService.cpp
)

# Link against the QuantumMeshBoost library
target_link_libraries(QuantumRenderService PRIVATE QuantumMeshBoost
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml.lib
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml-base.lib
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml-cpu.lib
)

# Include directories
target_include_directories(QuantumRenderService PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/include
)

# Set output directory for the executable
set_target_properties(QuantumRenderService PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/Release
)

# Copy GGML DLLs to output directory
if(WIN32)
    add_custom_command(TARGET QuantumRenderService POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml-base.dll
            $<TARGET_FILE_DIR:QuantumRenderService>
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml-cpu.dll
            $<TARGET_FILE_DIR:QuantumRenderService>
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml.dll
            $<TARGET_FILE_DIR:QuantumRenderService>
    )
endif()
