# AMD Quantum Mesh Rendering Boost Sample Application CMakeLists.txt
# Demonstrates how to use the quantum mesh rendering optimization

cmake_minimum_required(VERSION 3.10)

# Project information
project(QuantumMeshSampleApp VERSION 1.0.0 DESCRIPTION "AMD Quantum Mesh Rendering Boost Sample Application")

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add the sample application executable
add_executable(SampleApp
    SampleApp.cpp
)

# Add the complete system test executable
add_executable(TestCompleteSystem
    TestCompleteSystem.cpp
)

# Link against the QuantumMeshBoost library
target_link_libraries(SampleApp PRIVATE QuantumMeshBoost
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml.lib
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml-base.lib
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml-cpu.lib
)

target_link_libraries(TestCompleteSystem PRIVATE QuantumMeshBoost
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml.lib
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml-base.lib
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/src/Release/ggml-cpu.lib
)

# Include directories
target_include_directories(SampleApp PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/include
)

# Set output directory for the executable
set_target_properties(SampleApp PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/Release
)

set_target_properties(TestCompleteSystem PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/Release
)

# Copy GGML DLLs to output directory
if(WIN32)
    add_custom_command(TARGET SampleApp POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml-base.dll
            $<TARGET_FILE_DIR:SampleApp>
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml-cpu.dll
            $<TARGET_FILE_DIR:SampleApp>
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml.dll
            $<TARGET_FILE_DIR:SampleApp>
    )
    
    add_custom_command(TARGET TestCompleteSystem POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml-base.dll
            $<TARGET_FILE_DIR:TestCompleteSystem>
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml-cpu.dll
            $<TARGET_FILE_DIR:TestCompleteSystem>
        COMMAND ${CMAKE_COMMAND} -E copy
            ${CMAKE_CURRENT_SOURCE_DIR}/../ggml/build-vs/bin/Release/ggml.dll
            $<TARGET_FILE_DIR:TestCompleteSystem>
    )
endif()
